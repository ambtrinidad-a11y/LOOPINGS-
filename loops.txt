1. Write a program to calculate the factorial value of the input number n. Filename: factorial.cpp 

Sample output:

                         Enter a number: 4
                         Factorial value: 24

#include <iostream>

using namespace std;

int main() {
  int n;

  cout << "Enter a number: ";
  cin >> n;

  if (n < 0) {
    cout << "Factorial is not defined for negative numbers." << endl;
  } else {
    long long factorial = 1;
    if (n == 0) {
      factorial = 1;
    } else {
     int i = 1;
      while (i <= n) {
        factorial *= i;
        i++;
      }
    }
    cout << "Factorial value: " << factorial << endl;
  }
  return 0;
}



2.Fibonacchi
#include <iostream>

using namespace std;

int main() {
    int n;

    cout << "Enter a number: ";
    cin >> n;

    if (n <= 0) {
        cout << "Please enter a positive integer." << endl;
        return 1;
    }

    
    int fibonacci[n];

    if (n >= 1) {
        fibonacci[0] = 1;
    }
    if (n >= 2) {
        fibonacci[1] = 1;
    }

    for (int i = 2; i < n; ++i) {
        fibonacci[i] = fibonacci[i - 1] + fibonacci[i - 2];
    }

    cout << "Fibonacci series: ";
    for (int i = 0; i < n; ++i) {
        cout << fibonacci[i] << (i == n - 1 ? "" : "  ");
    }
    cout << endl;

    return 0;
}
 

3. REVERSE NUMBER 
#include <iostream>
using namespace std;

int main() {
    int n;

    cout << "Enter a number (1 to 5): ";
    cin >> n;

    if (n <= 0 || n > 5) {
        cout << "Please enter a positive integer between 1 and 5." << endl;
        return 1;
    }

    long long sum = 0;

    for (int i = 1; i <= n; ++i) {
        long long power = 1;

        // Calculate i raised to the power of n
        for (int j = 0; j < n; ++j) {
            power *= i;
        }

        sum += power;
    }

    cout << "The sum of powers from 1 to " << n << " is: " << sum << endl;

    return 0;
}
